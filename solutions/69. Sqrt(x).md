```python
def mySqrt(self, x):
    s = x
    while s * s > x:
        s = (s + x // s) // 2
    return s
```

```python
def mySqrt(self, x):
    l, r = 0, x
    while l < r:
        mid = (l + r + 1) // 2
        if mid * mid <= x:
            l = mid
        else:
            r = mid - 1
    return l
```
