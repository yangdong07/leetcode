```python
def reverseBetween(self, head, m, n):
    prev = dummy = ListNode(None)
    node = dummy.next = head
    index = 1

    while index < m:
        node, prev = node.next, node
        index += 1

    # node is @m, prev is @m-1
    m_prev = prev
    m_node = node

    while index <= n:
        prev, node.next, node = node, prev, node.next
        index += 1

    # node is @n+1, prev is @n
    m_prev.next = prev
    m_node.next = node

    return dummy.next
```
