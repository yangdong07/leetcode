```python
class Solution:
    def permute(self, nums):
        return [[n] + p
                for i, n in enumerate(nums)
                for p in self.permute(nums[:i] + nums[i + 1:])] or [[]]
```

```python
def permute(self, nums):
      permutations = [[]]
      for num in nums:
          permutations = [p[:i] + [num] + p[i:] for p in permutations for i in range(len(p) + 1)]
      return permutations
```
