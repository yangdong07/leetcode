#### Solution

```python
class Solution:
    def isPalindrome(self, x):
        """
        :type x: int
        :rtype: bool
        """
        if 0 <= x < 10:
            return True
        if x % 10 == 0:
            return False

        y = 0
        while (x > y):
            y = y * 10 + x % 10
            x = x // 10
            if y == x or y == x // 10:
                return True

        return False

print(Solution().isPalindrome(123454321))
```

```python
class Solution:
    def isPalindrome(self, x):
        """
        :type x: int
        :rtype: bool
        """
        s = str(x)
        return s == s[::-1]
```
