```python
def removeDuplicates(self, nums):
    if len(nums) < 3:
        return len(nums)
    i = 1
    for j in range(2, len(nums)):
        if nums[j] != nums[i] or nums[i] != nums[i - 1]:
            i += 1
            nums[i] = nums[j]
    return i + 1
```

```python
def removeDuplicates(self, nums):
    if len(nums) < 3:
        return len(nums)
    ans = nums[:2]
    for j in range(2, len(nums)):
        if nums[j] != ans[-1] or ans[-1] != ans[-2]:
            ans.append(nums[j])
    return len(ans)
```
