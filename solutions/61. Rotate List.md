```python
def rotateRight(self, head, k):
    if not head:
        return None

    n = 1
    cursor = head
    while cursor.next:
        n += 1
        cursor = cursor.next

    k = k % n
    if k == 0:
        return head

    cursor.next = head
    cursor = head
    for i in range(n - k - 1):
        cursor = cursor.next
    head = cursor.next
    cursor.next = None

    return head
```
